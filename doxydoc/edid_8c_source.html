89 void parse_mfg_id_in_buffer(Byte * mfg_id_bytes, char * result, int bufsize) {
90  assert(bufsize >= 4);
91  result[0] = (mfg_id_bytes[0] >> 2) & 0x1f;
92  result[1] = ((mfg_id_bytes[0] & 0x03) << 3) | ((mfg_id_bytes[1] >> 5) & 0x07);
93  result[2] = mfg_id_bytes[1] & 0x1f;
94  // printf("result[0] = 0x%02x\n", result[0]);

96  // printf("result[2] = 0x%02x\n", result[2]);
97  result[0] += 64;
98  result[1] += 64;
99  result[2] += 64;
100  result[3] = 0; // terminating null
101 }

143 static void get_edid_descriptor_strings(
144  Byte* edidbytes,
145  char* namebuf,
146  int namebuf_len,
147  char* snbuf,
148  int snbuf_len,
149  char* otherbuf,
150  int otherbuf_len)
151 {
152  bool debug = false;
153  assert(namebuf_len >= 14 && snbuf_len >= 14 && otherbuf_len >= 14);
154  strcpy(namebuf, "Unspecified");
155  strcpy(snbuf, "Unspecified");
156  strcpy(otherbuf, "Unspecified");
157 
158  int fields_found = 0;
159 
160  // 4 descriptor blocks beginning at offset 54. Each block is 18 bytes.
161  // In each block, bytes 0-3 indicates the contents.
162  int descriptor_ndx = 0;
163  for (descriptor_ndx = 0; descriptor_ndx < EDID_DESCRIPTOR_BLOCK_CT; descriptor_ndx++) {
164  Byte * descriptor = edidbytes +
165  EDID_DESCRIPTORS_BLOCKS_START +
166  descriptor_ndx * EDID_DESCRIPTOR_BLOCK_SIZE;
167  if (debug)
168  printf("(%s) full descriptor: %s\n", __func__,
169  hexstring(descriptor, EDID_DESCRIPTOR_BLOCK_SIZE));
170 
171  // test if a string descriptor
172  if ( descriptor[0] == 0x00 && // 0x00 if not a timing descriptor
173  descriptor[1] == 0x00 && // 0x00 if not a timing descriptor
174  descriptor[2] == 0x00 && // 0x00 for all descriptors
175  descriptor[4] == 0x00
176  )
177  {
178  char * nameslot = NULL;
179  switch(descriptor[3]) {
180  case 0xff: nameslot = snbuf; break; // monitor serial number
181  case 0xfe: nameslot = otherbuf; break; // arbitrary ASCII string
182  case 0xfc: nameslot = namebuf; break; // monitor name
183  }
184 
185  if (nameslot) {
186  Byte * textstart = descriptor+5;
187  // DBGMSF(debug, "String in descriptor: %s", hexstring(textstart, 14));
188  int offset = 0;
189  while (*(textstart+offset) != 0x0a && offset < 13) {
190  // DBGMSG("textlen=%d, char=0x%02x", textlen, *(textstart+textlen));
191  nameslot[offset] = *(textstart+offset);
192  offset++;
193  }
194  // memcpy(nameslot, textstart, offset);
195  nameslot[offset] = '\0';
196  if (debug)
197  printf("(%s) name = %s\n", __func__, nameslot);
198 
199  fields_found++;
200  }
201  }
202  }
203 }

214 Parsed_Edid * create_parsed_edid(Byte* edidbytes) {
215  assert(edidbytes);
216  bool debug = true;
217  bool ok = true;
218  Parsed_Edid* parsed_edid = NULL;
219 
220  const Byte edid_header_tag[] = {0x00, 0xff, 0xff, 0xff, 0xff, 0xff, 0xff, 0x00};
221  if (memcmp(edidbytes, edid_header_tag, 8) != 0) {
222  char * hs = hexstring(edidbytes,8);
223  if (debug)
224  printf("(%s) Invalid initial EDID bytes: %s\n", __func__, hs);
225  free(hs);
226  goto bye;
227  }
228 
229  if (edid_checksum(edidbytes) != 0x00) {
230  if (debug)
231  printf("(%s) Invalid EDID checksum: 0x%02x\n", __func__, edid_checksum(edidbytes));
232  goto bye;
233  }
234 
235  parsed_edid = calloc(1,sizeof(Parsed_Edid));
236  assert(sizeof(parsed_edid->bytes) == 128);
237  memcpy(parsed_edid->marker, EDID_MARKER_NAME, 4);
238  memcpy(parsed_edid->bytes, edidbytes, 128);
239 
240  get_edid_mfg_id_in_buffer(
241  edidbytes,
242  parsed_edid->mfg_id,
243  sizeof(parsed_edid->mfg_id) );
244 
245  parsed_edid->model_hex = edidbytes[0x0b] << 8 | edidbytes[0x0a];
246 
247  parsed_edid->serial_binary = edidbytes[0x0c] |
248  edidbytes[0x0d] << 8 |
249  edidbytes[0x0e] << 16 |
250  edidbytes[0x0f] << 24;
251 
252  get_edid_descriptor_strings(
253  edidbytes,
254  parsed_edid->model_name, sizeof(parsed_edid->model_name),
255  parsed_edid->serial_ascii, sizeof(parsed_edid->serial_ascii),
256  parsed_edid->extra_descriptor_string, sizeof(parsed_edid->extra_descriptor_string)
257  );
258 
259  parsed_edid->year = edidbytes[17] + 1990;
260  parsed_edid->is_model_year = edidbytes[16] == 0xff;
261  parsed_edid->edid_version_major = edidbytes[18];
262 #ifdef UNNEEDED
263  if (parsed_edid->edid_version_major != 1 && parsed_edid->edid_version_major != 2) {
264  DBGMSF(debug, "Invalid EDID major version number: %d", parsed_edid->edid_version_major);
265  ok = false;
266  }
267 #endif
268  parsed_edid->edid_version_minor = edidbytes[19];
269 
270  parsed_edid->rx = edidbytes[0x1b] << 2 | ( (edidbytes[0x19]&0b11000000)>>6 );
271  parsed_edid->ry = edidbytes[0x1c] << 2 | ( (edidbytes[0x19]&0b00110000)>>4 );
272  parsed_edid->gx = edidbytes[0x1d] << 2 | ( (edidbytes[0x19]&0b00001100)>>2 );
273  parsed_edid->gy = edidbytes[0x1e] << 2 | ( (edidbytes[0x19]&0b00000011)>>0 );
274  parsed_edid->bx = edidbytes[0x1f] << 2 | ( (edidbytes[0x1a]&0b11000000)>>6 );
275  parsed_edid->by = edidbytes[0x20] << 2 | ( (edidbytes[0x1a]&0b00110000)>>4 );
276  parsed_edid->wx = edidbytes[0x21] << 2 | ( (edidbytes[0x1a]&0b00001100)>>2 );
277 // parsed_edid->wy = edidbytes[0x22] << 2 | ( (edidbytes[0x1a]&0b00000011)>>0 );
278 // low order digits wrong, try another way
279  parsed_edid->wy = edidbytes[0x22] * 4 + ((edidbytes[0x1a]&0b00000011)>>0);
280 
281  parsed_edid->video_input_definition = edidbytes[0x14];
282  // printf("(%s) video_input_parms_bitmap = 0x%02x\n", __func__, video_input_parms_bitmap);
283  // parsed_edid->is_digital_input = (parsed_edid->video_input_definition & 0x80) ? true : false;
284  parsed_edid->extension_flag = edidbytes[0x7e];
285 
286  if (!ok) {
287  free(parsed_edid);
288  parsed_edid = NULL;
289  }
290 
291 bye:
292  return parsed_edid;
293 }

320 void report_parsed_edid_base(Parsed_Edid * edid, bool verbose, bool show_raw, int depth) {
321  int d1 = depth+1;
322  // verbose = true;
323  if (edid) {
324  rpt_vstring(depth,"EDID synopsis:");
325 
326  rpt_vstring(d1,"Mfg id: %s", edid->mfg_id);
327  rpt_vstring(d1,"Model: %s", edid->model_name);
328  rpt_vstring(d1,"Serial number: %s", edid->serial_ascii);
329  char * title = (edid->is_model_year) ? "Model year" : "Manufacture year";
330  rpt_vstring(d1,"%-16s: %d", title, edid->year);
331  rpt_vstring(d1,"EDID version: %d.%d", edid->edid_version_major, edid->edid_version_minor);
332 
333  if (verbose) {
334  rpt_vstring(d1,"Product code: 0x%04x (%u)", edid->model_hex, edid->model_hex);
335  // useless, binary serial number is typically 0x00000000 or 0x01010101
336  // rpt_vstring(d1,"Binary sn: %u (0x%08x)", edid->serial_binary, edid->serial_binary);
337  rpt_vstring(d1,"Extra descriptor: %s", edid->extra_descriptor_string);
338  char explbuf[100];
339  explbuf[0] = '\0';
340  if (edid->video_input_definition & 0x80) {
341  strcpy(explbuf, "Digital Input");
342  if (edid->edid_version_major == 1 && edid->edid_version_minor >= 4) {
343  switch (edid->video_input_definition & 0x0f) {
344  case 0x00:
345  strcat(explbuf, "(Digital interface not defined)");
346  break;
347  case 0x01:
348  strcat(explbuf, "(DVI)");
349  break;
350  case 0x02:
351  strcat(explbuf, "(HDMI-a)");
352  break;
353  case 0x03:
354  strcat(explbuf, "(HDMI-b");
355  break;
356  case 0x04:
357  strcat(explbuf, "(MDDI)");
358  break;
359  case 0x05:
360  strcat(explbuf, "(DisplayPort)");
361  break;
362  default:
363  strcat(explbuf, "(Invalid DVI standard)");
364  }
365  }
366  }
367  else {
368  strcpy(explbuf, "Analog Input");
369  }
370 
371  rpt_vstring(d1,"Video input definition: 0x%02x - %s", edid->video_input_definition, explbuf);
372  // rpt_vstring(d1,"Video input: %s", (edid->is_digital_input) ? "Digital" : "Analog");
373  rpt_vstring(d1,"White x,y: %.3f, %.3f", edid->wx/1024.0, edid->wy/1024.0);
374  rpt_vstring(d1,"Red x,y: %.3f, %.3f", edid->rx/1024.0, edid->ry/1024.0);
375  rpt_vstring(d1,"Green x,y: %.3f, %.3f", edid->gx/1024.0, edid->gy/1024.0);
376  rpt_vstring(d1,"Blue x,y: %.3f, %.3f", edid->bx/1024.0, edid->by/1024.0);
377  // restrict to EDID version >= 1.3?
378  rpt_vstring(d1,"Extension blocks: %u", edid->extension_flag);
379  }
380 
381  if (verbose) {
382  if (edid->edid_source)
383  rpt_vstring(depth,"EDID source: %s", edid->edid_source);
384  }
385  if (show_raw) {
386  rpt_vstring(depth,"EDID hex dump:");
387  rpt_hex_dump(edid->bytes, 128, d1);
388  }
389 
390  }
391  else {
392  // if (verbose)
393  rpt_vstring(d1,"No EDID");
394  }
395 }
