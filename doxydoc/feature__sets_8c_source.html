59 static struct _Vcp_Subset_Desc * find_subset_desc(VCP_Feature_Subset subset_id) {
60  struct _Vcp_Subset_Desc * result = NULL;
61  int ndx = 0;
62  for(; ndx<vcp_subset_count; ndx++) {
63  if (subset_id == vcp_subset_desc[ndx].subset_id) {
64  result = &vcp_subset_desc[ndx];
65  break;
66  }
67  }
68  assert(result);
69  return result;
70 }
