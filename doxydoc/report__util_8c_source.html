224 void rpt_vstring(int depth, char * format, ...) {
225  int buffer_size = 200;
226  char buffer[buffer_size];
227  char * buf = buffer;
228  va_list(args);
229  va_start(args, format);
230  int reqd_size = vsnprintf(buffer, buffer_size, format, args);
231  // if buffer wasn't sufficiently large, allocate a temporary buffer
232  if (reqd_size >= buffer_size) {
233  // printf("(%s) Allocating temp buffer, reqd_size=%d\n", __func__, reqd_size);
234  buf = malloc(reqd_size+1);
235  va_start(args, format);
236  vsnprintf(buf, reqd_size+1, format, args);
237  }
238  va_end(args);
239 
240  rpt_title(buf, depth);
241 
242  if (buf != buffer)
243  free(buf);
244 }

323 /* Reports the contents of a file.
324  *
325  * @param fn name of file
326  * @depth depth logical indentation depth

328 int rpt_file_contents(const char * fn, int depth) {
329  GPtrArray * line_array = g_ptr_array_new();
330  int rc = file_getlines(fn, line_array, false);
331  if (rc < 0) {
332  rpt_vstring(depth, "Error reading file %s: %s", fn, strerror(-rc));
333  }
334  else if (rc > 0) {
335  int ndx = 0;
336  for (; ndx < line_array->len; ndx++) {
337  char * curline = g_ptr_array_index(line_array, ndx);
338  rtrim_in_place(curline); // strip trailing newline
339  rpt_title(curline, depth);
340  }
341  }
342  return rc;
343 }
