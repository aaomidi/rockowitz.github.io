184 char * linux_errno_desc(int error_number) {
185  bool debug = false;
186  if (debug)
187  printf("(%s) error_number = %d\n", __func__, error_number);
188  assert(error_number >= 0);
189  Status_Code_Info * pdesc = find_errno_description(error_number);
190  if (pdesc) {
191  snprintf(workbuf, WORKBUF_SIZE, "%s(%d): %s",
192  pdesc->name, error_number, pdesc->description);
193  }
194  else {
195  snprintf(workbuf, WORKBUF_SIZE, "%d: %s",
196  error_number, strerror(error_number));
197  }
198  if (debug)
199  printf("(%s) error_number=%d, returning: |%s|\n", __func__, error_number, workbuf);
200  return workbuf;
201 }

210 /* Returns the Status_Code_Info record for the specified error number
211  *
212  * @param errnum linux error number, in positive, unmodulated form
213  *
214  * @return Status_Code_Description record, NULL if not found
215  *
216  * @remark
217  * If the description field of the found Status_Code_Info struct is NULL, it is set
218  * by calling strerror()
219  */
220 Status_Code_Info * find_errno_description(int errnum) {
221  bool debug = false;
222  if (debug)
223  printf("(%s) errnum=%d\n", __func__, errnum);
224  Status_Code_Info * result = NULL;
225  int ndx;
226  for (ndx=0; ndx < errno_desc_ct; ndx++) {
227  if (errnum == errno_desc[ndx].code) {
228  result = &errno_desc[ndx];
229  break;
230  }
231  }
232  if (result && !result->description) {
233  // char * desc = strerror(errnum);
234  // result->description = malloc(strlen(desc)+1);
235  // strcpy(result->description, desc);
236  result->description = strdup(strerror(errnum));
237  }
238  if (debug)
239  printf("(%s) Returning %p\n", __func__, result);
240  return result;
241 }

245 Status_Code_Info * create_dynamic_errno_info(int errnum) {
246  Status_Code_Info * result = &dummy_errno_desc;
247  result->code = errnum;

250  // would be simpler to use strerror_r(), but the return value of

254  strncpy(dummy_errno_description, s, sz);
255  dummy_errno_description[sz-1] = '\0'; // ensure trailing null in case strncpy truncated
256  result->description = dummy_errno_description;

285 bool errno_name_to_number(const char * errno_name, int * perrno) {
286  int found = false;
287  *perrno = 0;
288  for (int ndx = 0; ndx < errno_desc_ct; ndx++) {
289  if ( streq(errno_desc[ndx].name, errno_name) ) {
290  *perrno = -errno_desc[ndx].code;
291  found = true;
292  break;
293  }
294  }
295  return found;
296 }
