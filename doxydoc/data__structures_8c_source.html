488 bool store_bytehex_list(char * start, int len, void * data_struct, Byte_Appender appender){
489  bool ok = true;
490 
491  char * buf = malloc(len+1);
492  memcpy(buf, start, len);
493  buf[len] = '\0';
494 
495  char * curpos = buf;
496  char * nexttok;
497  Byte byteVal = 0x00; // initialization logically unnecessary, but makes compiler happy
498  while ( (nexttok = strtok(curpos, " ")) != NULL) {
499  if (curpos)
500  curpos = NULL; // for all calls after first
501  int ln = strlen(nexttok);
502  bool hexok = false;
503  if (ln == 2) { // normal case
504  // byteVal = hhc_to_byte(nexttok);
505  hexok = hhc_to_byte_in_buf(nexttok, &byteVal);
506  }
507  else if (ln == 1) {
508  // on old ultrasharp connected to blackrock (pre v2), values in capabilities
509  // string are single digits. Not clear whether to regard them as decimal or hex,
510  // since all values are < 9. But in that case decimal and single digit hex
511  // give the same value.
512  char buf[2];
513  buf[0] = '0';
514  buf[1] = *nexttok;
515  // byteVal = hhc_to_byte(buf);
516  hexok = hhc_to_byte_in_buf(buf, &byteVal);
517  }
518  if (!hexok) {
519  printf("(%s) Invalid hex value in list: %s\n", __func__, nexttok);
520  ok = false;
521  }
522  // printf("(%s) byteVal=0x%02x \n", __func__, byteVal );
523  if (hexok)
524  appender(data_struct, byteVal);
525  }
526 
527  free(buf);
528  // printf("(%s) Returning %s\n", __func__, bool_repr(ok));
529  return ok;
