243 static
244 int log_any_status_code(Status_Code_Counts * pcounts, int rc, const char * caller_name) {
245  bool debug = false;
246  DBGMSF(debug, "caller=%s, rc=%d", caller_name, rc);
247  assert(pcounts->error_counts_hash);
248  pcounts->total_status_counts++;
249 
250  if (rc == 0) {
251  DBGMSG("Called with rc = 0, from function %s", caller_name);
252  }
253 
254  // n. if key rc not found, returns NULL, which is 0
255  int ct = GPOINTER_TO_INT(g_hash_table_lookup(pcounts->error_counts_hash, GINT_TO_POINTER(rc)) );
256  g_hash_table_insert(pcounts->error_counts_hash, GINT_TO_POINTER(rc), GINT_TO_POINTER(ct+1));
257  // DBGMSG("Old count=%d", ct);
258 
259  // check the new value
260  int newct = GPOINTER_TO_INT(g_hash_table_lookup(pcounts->error_counts_hash, GINT_TO_POINTER(rc)) );
261  // DBGMSG("new count for key %d = %d", rc, newct);
262  assert(newct == ct+1);
263 
264  return ct+1;
265 }

483 void call_tuned_sleep(DDCA_IO_Mode io_mode, Sleep_Event_Type event_type) {
484  int sleep_time_millis = 0; // should be a default
485  switch(io_mode) {
486 
487  case DDC_IO_DEVI2C:
488  switch(event_type) {
489  case (SE_WRITE_TO_READ):
490  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
491  switch(sleep_strategy) {
492  case (1):
493  sleep_time_millis = sleep_time_millis/2;
494  break;
495  case (2):
496  sleep_time_millis = sleep_time_millis*2;
497  break;
498  default:
499  break;
500  }
501  break;
502  case (SE_POST_WRITE):
503  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
504  switch(sleep_strategy) {
505  case (1):
506  sleep_time_millis = sleep_time_millis/2;
507  break;
508  case (2):
509  sleep_time_millis = sleep_time_millis*2;
510  break;
511  default:
512  break;
513  }
514  break;
515  case (SE_POST_OPEN):
516  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
517  break;
518  case (SE_POST_READ):
519  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
520  break;
521  default:
522  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
523  } // switch within DDC_IO_DEVI2C
524  break;
525 
526  case DDC_IO_ADL:
527  switch(event_type) {
528  case (SE_WRITE_TO_READ):
529  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
530  break;
531  case (SE_POST_WRITE):
532  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
533  break;
534  case (SE_POST_OPEN):
535  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
536  break;
537  default:
538  sleep_time_millis = DDC_TIMEOUT_MILLIS_DEFAULT;
539  }
540  break;
541 
542  case USB_IO:
543  printf("(%s) call_tuned_sleep() called for USB_IO\n", __func__);
544  break;
545 
546  }
