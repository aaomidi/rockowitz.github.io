141 /* Finds the ADL library and dynamically loads it
142  *
143  * Arguments:
144  * pLocAdl_Procs return function table here
145  * verbose if true, verbose messages

147  * Returns:
148  * 0 loaded

152 static int link_adl(Adl_Procs** pLocAdl_Procs, bool verbose) {
153  bool debug = false;
154  DBGMSF(debug, "Starting." );
155  Adl_Procs* adl = calloc(1, sizeof(Adl_Procs));
156  int result = 0;
157 
158 #define LOADFUNC(_n_) \
159  do { \
160  DBGMSF(debug, "Loading function %s", #_n_); \
161  adl->_n_ = dlsym(adl->hDLL, #_n_); \
162  if (!adl->_n_) { \
163  SEVEREMSG("ADL error: loading symbol %s\n", #_n_); \
164  result = -1; \
165  } \
166  } while (0)
167 
168 
169  adl->hDLL = dlopen("libatiadlxx.so", RTLD_LAZY|RTLD_GLOBAL);
170  if (!adl->hDLL) {
171  if (verbose)
172  printf("ADL library libatiadlxx.so not found.\n" ); // this is a user error msg
173  result = 1;
174  }
175  else {
176  LOADFUNC(ADL_Main_Control_Create);
177  LOADFUNC(ADL_Main_Control_Destroy);
178 
179  LOADFUNC(ADL_Adapter_NumberOfAdapters_Get);
