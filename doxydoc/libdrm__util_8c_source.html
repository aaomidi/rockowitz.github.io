538 void report_property_value(
539  int fd,
540  drmModePropertyRes * prop_ptr,
541  uint64_t prop_value,
542  int depth)
543 {
544  int d1 = depth+1;
545  rpt_vstring(depth, "Property id: %d", prop_ptr->prop_id);
546  rpt_vstring(d1, "Name: %s", prop_ptr->name);
547  rpt_vstring(d1, "Flags: 0x%04x - %s", prop_ptr->flags,
548  interpret_property_flags(prop_ptr->flags) );
549  rpt_vstring(d1, "prop_value: %d 0x%08x", prop_value, prop_value);
550 
551  if (prop_ptr->flags & DRM_MODE_PROP_ENUM) {
552  for (int i = 0; i < prop_ptr->count_enums; i++) {
553  if (prop_ptr->enums[i].value == prop_value) {
554  rpt_vstring(d1, "Property value(enum) = %d - %s", prop_value, prop_ptr->enums[i].name);
555  break;
556  }
557  }
558  }
559 
560  else if (prop_ptr->flags & DRM_MODE_PROP_BITMASK) {
561  char buf[200] = ""; int bufsz=200;
562  bool not_truncated = true;
563  for (int i = 0; i < prop_ptr->count_enums && not_truncated; i++) {
564  if (prop_ptr->enums[i].value & prop_value) {
565  not_truncated = sbuf_append(buf, bufsz, ", ", prop_ptr->enums[i].name);
566  }
567  }
568  rpt_vstring(d1, "Property value(bitmask) = 0x%04x - %s", prop_value, buf);
569  }
570 
571  else if (prop_ptr->flags & DRM_MODE_PROP_RANGE) {
572  if (prop_ptr->count_values != 2) {
573  rpt_vstring(d1, "Property value = %d, Missing min or max value", prop_value);
574  }
575  else {
576  rpt_vstring(d1, "Property value(range) = %d, min=%d, max=%d",
577  prop_value, prop_ptr->values[0], prop_ptr->values[1]);
578  }
579  }
580 
581  // if (prop_ptr->flags & DRM_MODE_PROP_BLOB) {
582  else if (drm_property_type_is(prop_ptr, DRM_MODE_PROP_BLOB)) {
583  drmModePropertyBlobPtr blob_ptr = drmModeGetPropertyBlob(fd, prop_value);
584  if (!blob_ptr) {
585  rpt_vstring(d1, "Blob not found");
586  }
587  else {
588  report_drmModePropertyBlob(blob_ptr, d1);
589  drmModeFreePropertyBlob(blob_ptr);
590  }
591  }
592 
593  else if (drm_property_type_is(prop_ptr, DRM_MODE_PROP_OBJECT)) {
594  rpt_vstring(d1, "Object type, name = %s, value=%d",
595  prop_ptr->name, prop_value);
596  }
