308 // Exactly 1 of DDCA_RO, DDCA_WO, DDCA_RW is set
309 #define DDCA_RO 0x0400
310 #define DDCA_WO 0x0200
311 #define DDCA_RW 0x0100
312 #define DDCA_READABLE (DDCA_RO | DDCA_RW)
313 #define DDCA_WRITABLE (DDCA_WO | DDCA_RW)
315 // Further refine the C/NC/TABLE categorization of the MCCS spec
316 // Exactly 1 of the following 7 bits is set
317 #define DDCA_STD_CONT 0x80
318 #define DDCA_COMPLEX_CONT 0x40
319 #define DDCA_SIMPLE_NC 0x20
320 #define DDCA_COMPLEX_NC 0x10
321 // For WO NC features. There's no interpretation function or lookup table
322 // Used to mark that the feature is defined for a version
323 #define DDCA_WO_NC 0x08 // TODO: CHECK USAGE
324 #define DDCA_READABLE_TABLE 0x04
325 #define DDCA_WO_TABLE 0x02
327 #define DDCA_CONT (DDCA_STD_CONT|DDCA_COMPLEX_CONT)
328 #define DDCA_NC (DDCA_SIMPLE_NC|DDCA_COMPLEX_NC|DDCA_WO_NC)
329 #define DDCA_NON_TABLE (DDCA_CONT | DDCA_NC)
331 #define DDCA_TABLE (DDCA_READABLE_TABLE | DDCA_WO_TABLE)
