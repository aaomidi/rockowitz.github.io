44 /*
45 
46 Notes on status code management.
47 
48 Status codes in the ddcutil have multiple sources:
49 
50 1) Linux system calls.
51 - In general, functions return 0 or a positive value to indicate success.
52 - Values greater than 0 indicate something about a successful call, such as the number of bytes read.
53 - Negative values indicate that an error occurred. In that case the special variable errno is
54  typically the "error code", though some packages set the return code to -errno.
55 - Errno values, listed in errno.h, are positive numbers ranging from 1 to apparently
56  less than 200.
57 
58 2) ADL
59 - ADL functions return status codes, listed in file adl_defines.h
60 - The value of these codes ranges from 4 to -21. 0 indicates normal success.
61 - Positive values appear to be "ok, but". Not clear when these values occur or what to do about them.
62 - Negative values indicate errors, some of which may reflect programming errors.
63 
64 3) ddcutil specific status codes.
65 
66 
67 Problem: Linux and ADL error numbers conflict.
68 ddcutil error numbers can be assigned to a range out of conflict.
69 
70 Solution.
71 
72 Mulitplexing.
73 
74  */

185 int modulate_rc(int rc, Retcode_Range_Id range_id){
186  bool debug = false;
187  if (debug)
188  printf("(%s) rc=%d, range_id=%d\n", __func__, rc, range_id);
189  assert( abs(rc) <= RCRANGE_BASE_MAX );
190  int base = retcode_range_table[range_id].base;
191  if (rc != 0) {
192  if (rc < 0)
193  rc -= base;
194  else
195  rc += base;
196  }
197  if (debug)
198  printf("(%s) Returning: %d\n", __func__, rc);
199  return rc;
200 }

215  // TODO: check that rc is in the specified modulation range

234 Retcode_Range_Id get_modulation(int rc) {
235  int ndx = 0;
236  int abs_rc = abs(rc);
237  Retcode_Range_Id range_id;
238  for (;ndx < retcode_range_ct; ndx++) {
239  if (abs_rc >= retcode_range_table[ndx].base && abs_rc <= retcode_range_table[ndx].max) {
240  range_id = retcode_range_table[ndx].id;
241  assert (range_id == ndx);
242  break;
243  }
244  }
245  assert(ndx < retcode_range_ct); // fails if not found
246  return range_id;
247 }

284 Status_Code_Info * find_global_status_code_info(Global_Status_Code gsc) {
285  bool debug = false;
286  // use don't use DBGMSG to avoid circular includes
287  if (debug)
288  printf("(%s) Starting. rc = %d\n", __func__, gsc);
289 
290  Status_Code_Info * pinfo = NULL;
291 
292  if (gsc == 0)
293  pinfo = &ok_status_code_info;
294  else {
295  Retcode_Range_Id modulation = get_modulation(gsc);
296  if (debug)
297  printf("(%s) modulation=%d\n", __func__, modulation);
298 
299  Retcode_Description_Finder finder_func = retcode_range_table[modulation].desc_finder;
300  assert(finder_func != NULL);
301  bool finder_arg_is_modulated = retcode_range_table[modulation].finder_arg_is_modulated;
302  int rawrc = (finder_arg_is_modulated) ? gsc : demodulate_rc(gsc, modulation);
303  if (debug)
304  printf("(%s) rawrc = %d\n", __func__, rawrc);
305  pinfo = finder_func(rawrc);
306  }
307  if (debug) {
308  printf("(%s) Done. Returning %p", __func__, pinfo);
309  if (pinfo)
310  report_status_code_info(pinfo);
311  }
312 
313  return pinfo;
314 }
