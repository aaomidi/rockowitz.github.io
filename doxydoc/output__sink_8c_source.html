104 int printf_sink(Output_Sink sink, const char * format, ...) {
105  struct Output_Sink * psink = (struct Output_Sink*) sink;
106  assert(psink && memcmp(psink->marker, OUTPUT_SINK_MARKER, 4) == 0);
107  int rc = 0;
108  va_list(args);
109  va_start(args, format);
110  switch(psink->sink_type) {
111  case (SINK_STDOUT):
112  // rc = vprintf(format, args);
113  // break;
114  case (SINK_FILE):
115  rc = vfprintf(psink->fp, format, args);
116  if (rc < 0)
117  rc = -errno;
118  break;
119  case (SINK_MEMORY):
120  {
121  bool done = false;
122  while (!done) {
123  rc = vsnprintf(psink->workbuf, psink->cur_max_chars, format, args);
124  if (rc > psink->cur_max_chars) {
125  // if work buffer was too small, reallocate and retry
126  free(psink->workbuf);
127  psink->cur_max_chars = rc + 1;
128  psink->workbuf = calloc( (psink->cur_max_chars)+1, sizeof(char));
129  }
130  else
131  done = true;
132  };
133  if (rc < 0)
134  rc = -errno;
135  else
136  g_ptr_array_add(psink->line_array, strdup(psink->workbuf));
137  break;
138  }
139  }
140  va_end(args);
141  return rc;
142 }

174 int close_sink(Output_Sink sink) {
175  struct Output_Sink * psink = (struct Output_Sink *) sink;
176  assert(psink && memcmp(psink->marker, OUTPUT_SINK_MARKER, 4) == 0);
177  int rc = 0;
178 
179  switch(psink->sink_type) {
180  case (SINK_STDOUT):
181  break;
182  case (SINK_FILE):
183  rc = fclose(psink->fp);
184  if (rc < 0)
185  rc = -errno;
186  break;
187  case (SINK_MEMORY):
188  g_ptr_array_free(psink->line_array, true);
189  psink->line_array = NULL;
190  break;
191  }
192  psink->marker[3] = 'x';
193  free(psink);
194  return rc;
