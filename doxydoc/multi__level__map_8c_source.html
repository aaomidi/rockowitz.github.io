80 MLM_Node * mlm_add_node(Multi_Level_Map * map, MLM_Node * parent, uint key, char * value) {
81  // printf("(%s) parent=%p, key=0x%04x, value=|%s|\n", __func__, parent, key, value);
82  MLM_Node * new_node = calloc(1,sizeof(MLM_Node));
83  new_node->code = key;
84  new_node->name = value;
85  new_node->children = NULL;
86 
87  if (!parent) {
88  new_node->level = 0;
89  g_ptr_array_add(map->root, new_node);
90  }
91  else {
92  new_node->level = parent->level+1;
93  if (!parent->children) {
94  int initial_size = map->level_detail[parent->level].initial_size;
95  parent->children = g_ptr_array_sized_new(initial_size);
96  }
97  g_ptr_array_add(parent->children, new_node);
98  }
99  map->level_detail[new_node->level].total_entries += 1;
100  return new_node;
101 }

182 MLM_Node * mlm_find_child(GPtrArray * nodelist, uint id) {
183  bool debug = false;
184  if (debug)
185  printf("(%s) Starting, id=0x%08x\n", __func__, id);
186  MLM_Node * result = NULL;
187 
188  for (int ndx = 0; ndx < nodelist->len; ndx++) {
189  MLM_Node * cur_entry = g_ptr_array_index(nodelist, ndx);
190  if (debug) printf("(%s) Comparing code=0x%04x, name=%s\n",
191  __func__, cur_entry->code, cur_entry->name);
192  if (cur_entry->code == id) {
193  result = cur_entry;
194  break;
195  }
196  }
197 
198  if (debug)
199  printf("(%s) Returning %p\n", __func__, result);
200  return result;
201 }

224 Multi_Level_Names mlm_get_names2(Multi_Level_Map * mlm, int levelct, uint* ids) {
225  bool debug = false;
226  assert(levelct >= 1 && levelct <= MLT_MAX_LEVELS);
227  // pciusb_id_ensure_initialized(); // <== WHAT TO DO?
228  if (debug) {
229  printf("(%s) levelct=%d\n", __func__, levelct);
230  for (int ndx = 0; ndx < levelct; ndx++) {
231  printf(" ids[%d] = 0x%08x\n", ndx, ids[ndx]);
232  }
233  }
234 
235  Multi_Level_Names result = {0};
236 
237  int argndx = 0;
238  GPtrArray * children = mlm->root;
239  result.levels = 0;
240  while (argndx < levelct) {
241  // printf("(%s) argndx=%d\n", __func__, argndx);
242  if (!children)
243  break;
244  MLM_Node * level_entry = mlm_find_child(children, ids[argndx]);
245  if (!level_entry) {
246  break;
247  }
248  result.levels = argndx+1;
249  result.names[argndx] = level_entry->name;
250  children = level_entry->children;
251  argndx++;
252  }
253 
254  if (debug) {
255  printf("(%s) Returning: \n", __func__);
256  report_multi_level_names(&result, 1);
257  }
258  return result;
