105 /* Reports a single entry in the error simulation table,
106  * i.e. the conditions under which an error will be simulated
107  * for the function and the error value for the function to return.
108  *
109  * Arguments:
110  * key_ptr pointer to function name
111  * value_ptr pointer to GArray of Fsim_Call_Occ_Rec's
112  * user_data_ptr pointer to logical indentation depth
113  */
114 static void report_error_table_entry(
115  gpointer key_ptr,
116  gpointer value_ptr,
117  gpointer user_data_ptr)
118 {
119  bool debug = false;
120  if (debug)
121  printf("(%s) Starting. value_ptr=%p, user_data_ptr=%p\n", __func__, value_ptr, user_data_ptr);
122  char * key = (char *) key_ptr;
123  Fsim_Func_Rec * frec = (Fsim_Func_Rec *) value_ptr;
124  int * depth_ptr = (int *) user_data_ptr;
125  int depth = *depth_ptr;
126 
127  rpt_vstring(depth, "function: %s", key);
128  for (int ndx = 0; ndx < frec->call_occ_recs->len; ndx++) {
129  Fsim_Call_Occ_Rec occ_rec = g_array_index(frec->call_occ_recs, Fsim_Call_Occ_Rec, ndx);
130  rpt_vstring(depth+1, "rc = %d, occurrences=(%s, %d)",
131  occ_rec.rc,
132  (occ_rec.call_occ_type == FSIM_CALL_OCC_RECURRING) ? "recurring" : "single",
133  occ_rec.occno);
134  }
135 
136 }

158 /* Gets the record for a function in the failure simulation table.
159  * If a record does not already exist, a new one is created.
160  *
161  * Arguments:
162  * funcname function mame
163  *
164  * Returns: pointer to Fsim_Func_Rec for the function
165  */
166 static Fsim_Func_Rec * fsim_get_or_create_func_rec(char * funcname) {
167  GHashTable * fst = fsim_get_or_create_failsim_table();
168  Fsim_Func_Rec * frec = g_hash_table_lookup(fst, funcname);
169  if (!frec) {
170  frec = calloc(1, sizeof(Fsim_Func_Rec));
171  memcpy(frec->marker, FSIM_FUNC_REC_MARKER, 4);
172  frec->func_name = strdup(funcname);
173  frec->callct = 0;
174  frec->call_occ_recs = g_array_new(
175  false, // zero_terminated
176  true, // clear_
177  sizeof(Fsim_Call_Occ_Rec) );
178  g_hash_table_insert(fst, frec->func_name, frec);
179  }
180  return frec;
181 }

271 /* Evaluates a string status code expression.
272  *
273  * The string can take the following forms:
274  * integer e.g. "-42"
275  * boolean i.e. "true" or "false"
276  * a symbolic status code name, optionally prefixed by
277  * "modulated:" or "base:".
278  * If neither "modulated" nor "vase" is specified, "modulated" is assumed
279  * e.g. "DDC_RC_ALL_ZERO", "base:EBUSY"
280  *
281  * Arguments:
282  * rc_string string to evaluate
283  * evaluated_rc where to return evaluated value
284  *
285  * Returns: true if evaluation successful, false if error
286  */
287 
288 // Issue: should unmodulated values be negative, or should
289 // an optional minus sign be recognized?
290 // e.g. should we specify -EBUSY for -22?
291 bool eval_fsim_rc(char * rc_string, int * evaluated_rc) {
292  bool debug = true;
293  if (debug)
294  printf("(%s) Starting. rc_string=%s\n", __func__, rc_string);
295  char * end;
296  bool ok = false;
297  *evaluated_rc = 0;
298  long int answer = strtol(rc_string, &end, 10);
299  if (*end == '\0') {
300  *evaluated_rc = (int) answer;
301  ok = true;
302  }
303  else if (streq(rc_string, "true")) {
304  *evaluated_rc = true;
305  ok = true;
306  }
307  else if (streq(rc_string, "false")) {
308  *evaluated_rc = false;
309  ok = true;
310  }
311  else {
312  bool is_modulated = true;
313  if (str_starts_with(rc_string, "modulated:")) {
314  is_modulated = true;
315  rc_string = rc_string + strlen("modulated:");
316  }
317  else if (str_starts_with(rc_string, "base:")) {
318  is_modulated = false;
319  rc_string = rc_string + strlen("base:");
320  }
321  if (strlen(rc_string) == 0)
322  ok = false;
323  else if (is_modulated && name_to_number_func)
324  ok = name_to_number_func(rc_string, evaluated_rc);
325  else if (!is_modulated && unmodulated_name_to_number_func)
326  ok = unmodulated_name_to_number_func(rc_string, evaluated_rc);
327  else
328  ok = false;
329  }
330 
331  if (debug)
332  printf("(%s) Starting. rc_string=%s. Returning: %s, *evaluated_rc=%d\n",
333  __func__, rc_string, bool_repr(ok), *evaluated_rc);
334  return ok;
335 }

492 Failsim_Result fsim_check_failure(const char * fn, const char * funcname) {
493  bool debug = false;
494  Failsim_Result result = {false, 0};
495  if (fst) {
496  Fsim_Func_Rec * frec = g_hash_table_lookup(fst, funcname);
497  if (frec) {
498  frec->callct++;
499  for (int ndx = 0; ndx < frec->call_occ_recs->len; ndx++) {
500  Fsim_Call_Occ_Rec * occ_rec = &g_array_index(frec->call_occ_recs, Fsim_Call_Occ_Rec, ndx);
501  if (debug)
502  printf("(%s) call_occ_type=%d, callct = %d, occno=%d\n", __func__,
503  occ_rec->call_occ_type, frec->callct, occ_rec->occno);
504  if (occ_rec->call_occ_type == FSIM_CALL_OCC_RECURRING) {
505 
506  if ( frec->callct % occ_rec->occno == 0) {
507  result.force_failure = true;
508  result.failure_value = occ_rec->rc;
509  break;
510  }
511  }
512  else {
513  if (frec->callct == occ_rec->occno) {
514  result.force_failure = true;
515  result.failure_value = occ_rec->rc;
516  break;
517  }
518  }
519  }
520  if (result.force_failure) {
521  printf("Simulating failure for call %d of function %s, returning %d\n",
522  frec->callct, funcname, result.failure_value);
523  // printf("Call stack:\n");
524  // why wasn't this here in the original version?
525  show_backtrace(2);
526  }
527 
528  }
529  }
530 
531  if (debug)
532  printf("(%s) funcname=%s, returning (%d,%d)\n",
533  __func__, funcname, result.force_failure, result.failure_value);
534  return result;
535 }
