130 void report_udev_device(struct udev_device * dev, int depth) {
131  int d1 = depth+1;
132  int d2 = depth+2;
133  rpt_structure_loc("struct udev_device", dev, depth);
134  rpt_vstring(d1, "devpath: %s", udev_device_get_devpath(dev));
135  rpt_vstring(d1, "subsystem: %s", udev_device_get_subsystem(dev));
136  rpt_vstring(d1, "devtype: %s", udev_device_get_devtype(dev));
137  rpt_vstring(d1, "syspath: %s", udev_device_get_syspath(dev));
138  rpt_vstring(d1, "sysname: %s", udev_device_get_sysname(dev));
139  rpt_vstring(d1, "sysnum: %s", udev_device_get_sysnum(dev));
140  rpt_vstring(d1, "devnode: %s", udev_device_get_devnode(dev));
141 
142  struct udev_list_entry * cur_entry;
143  struct udev_list_entry * properties_list_entry;
144  struct udev_list_entry * sysattr_list_entry;
145 
146  properties_list_entry = udev_device_get_properties_list_entry(dev);
147  sysattr_list_entry = udev_device_get_sysattr_list_entry(dev);
148 
149  rpt_vstring(d1, "Properties:");
150  udev_list_entry_foreach(cur_entry, properties_list_entry) {
151  const char * prop_name = udev_list_entry_get_name(cur_entry);
152  const char * prop_value = udev_list_entry_get_value(cur_entry);
153  const char * prop_value2 = udev_device_get_property_value(dev, prop_name);
154  assert(streq(prop_value, prop_value2));
155  rpt_vstring(d2, "%s -> %s", prop_name, prop_value);
156  }
157 
158  rpt_vstring(d1, "Sysattrs:");
159  udev_list_entry_foreach(cur_entry, sysattr_list_entry) {
160  const char * attr_name = udev_list_entry_get_name(cur_entry);
161  const char * attr_value = udev_list_entry_get_value(cur_entry);
162  const char * attr_value2 = udev_device_get_sysattr_value(dev, attr_name);
163  assert(attr_value == NULL);
164  // hex_dump( (Byte*) attr_value2, strlen(attr_value2)+1);
165  if (attr_value2 && strchr(attr_value2, '\n')) {
166  // if (streq(attr_name, "uevent")) {
167  // output is annoying to visually scan since it contains newlines
168  Null_Terminated_String_Array ntsa = strsplit(attr_value2, "\n");
169  if (ntsa_length(ntsa) == 0)
170  rpt_vstring(d2, "%s -> %s", attr_name, "");
171  else {
172  rpt_vstring(d2, "%s -> %s", attr_name, ntsa[0]);
173  int ndx = 1;
174  while (ntsa[ndx]) {
175  rpt_vstring(d2, "%*s %s", strlen(attr_name) + 3, " ", ntsa[ndx]);
176  ndx++;
177  }
178  }
179  ntsa_free(ntsa);
180 
181 #ifdef ALTERNATIVE
182  // simpler, works
183  char * av = strdup(attr_value2);
184  char * p = av;
185  while (*p) {
186  if (*p == 0x0a)
187  *p = ',';
188  p++;
189  }
190  rpt_vstring(d2, "%s -> %s", attr_name, av);
191  free(av);
192 #endif
193  }
194  // n. attr_name "descriptors" returns a hex value, not a null-terminated string
195  // should display as hex, but how to determine length?
196  // for example of reading, see http://fossies.org/linux/systemd/src/udev/udev-builtin-usb_id.c
197  // not worth pursuing
198  else {
199  rpt_vstring(d2, "%s -> %s", attr_name, attr_value2);
200  }
201  }
202 }
