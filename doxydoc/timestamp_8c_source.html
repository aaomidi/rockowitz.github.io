62 long cur_realtime_nanosec() {
63  struct timespec tvNow;
64  clock_gettime(CLOCK_REALTIME, &tvNow);
65  // long result = (tvNow.tv_sec * 1000) + (tvNow.tv_nsec / (1000 * 1000) ); // milliseconds
66  // long result = (tvNow.tv_sec * 1000 * 1000) + (tvNow.tv_nsec / 1000); // microseconds
67  long result = tvNow.tv_sec * (1000 * 1000 * 1000) + tvNow.tv_nsec; // NANOSEC
68  if (tracking_timestamps && timestamp_ct < MAX_TIMESTAMPS) {
69  if (!timestamp_history) {
70  timestamp_ct = 0;
71  timestamp_history = calloc(MAX_TIMESTAMPS, sizeof(long));
72  }
73  timestamp[timestamp_ct++] = result;
74  }
75  // printf("(%s) Returning: %ld\n", result);
76  return result;
77 }
