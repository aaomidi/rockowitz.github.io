77 Status_Code_Info * ddcrc_find_status_code_info(int rc) {
78  Status_Code_Info * result = NULL;
79 
80  int ndx;
81  for (ndx=0; ndx < ddcrc_desc_ct; ndx++) {
82  if (rc == ddcrc_info[ndx].code) {
83  result = &ddcrc_info[ndx];
84  break;
85  }
86  }
87  return result;
88 }

90 /* Status code classification
91 
92  DDCRC_NULL_RESPONSE
93  DDCRC_ALL_TRIES_ZERO
94  DDCRC_REPORTED_UNSUPPORTED
95  DDCRC_DETERMINED_UNSUPPORTED
96 
97  DDCRC_REPORTED_UNSUPPORTED is a primary error, but reports a state, not really an error
98 
99  Derived codes are set after function has examined a primary code
100  Do not count as DDC errors.
101  Derived:
102  DDCRC_ALL_TRIES_ZERO
103  DDCRC_RETRIES
104  DDCRC_DETERMINED_UNSUPPORTED
105 
106  DDCRC NULL_RESPONSE is ambiguous
107  can be expected (DDC detection)
108  no answer to give, e.g. because not ready, not expected (protocol error)
109  but also is used by some monitors to indicate invalid request (e.g. unsupported VCP code)
110 
111  All others indicate real, primary errors
112 
113 
114  Issues:
115  - DDCRC_REPORTED_UNSUPPORTED should not be a fatal failure in try_stats,
116  it is a successful try, it's just that the response is "unsupported"
117  *
118  */

138 char * ddcrc_desc(int rc) {
139  char * result = NULL;
140  Status_Code_Info * pdesc = ddcrc_find_status_code_info(rc);
141  if (pdesc) {
142  snprintf(workbuf, 200,
143  "%s(%d): %s", pdesc->name, rc, pdesc->description);
144  }
145  else {
146  snprintf(workbuf, 200, "Unexpected status code %d", rc);
147  }
148  result = workbuf;
149  return result;
150 }

153 /* Gets the ddcutil error number for a symbolic name.
154  *
155  * Arguments:
156  * errno_name symbolic name, e.g. EBUSY
157  * perrno where to return error number
158  *
159  * Returns: true if found, false if not
160  */
161 bool ddc_error_name_to_number(const char * errno_name, Global_Status_Code * perrno) {
162  int found = false;
163  *perrno = 0;
164  for (int ndx = 0; ndx < ddcrc_desc_ct; ndx++) {
165  if ( streq(ddcrc_info[ndx].name, errno_name) ) {
166  *perrno = ddcrc_info[ndx].code;
167  found = true;
168  break;
169  }
170  }
171  return found;
172 }
