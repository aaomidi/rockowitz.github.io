50 /* Tests if a Version_Spec value represents a valid MCCS version,
51  * i.e. 1.0, 2.0, 2.1, 3.0, or 2.2.
52  *
53  * Arguments:
54  * vspec value to test
55  *
56  * Returns: true/false
57  */
58 static bool is_known_vcp_spec(DDCA_MCCS_Version_Spec vspec) {
59  bool result = vcp_version_eq(vspec, VCP_SPEC_V10) ||
60  vcp_version_eq(vspec, VCP_SPEC_V20) ||
61  vcp_version_eq(vspec, VCP_SPEC_V21) ||
62  vcp_version_eq(vspec, VCP_SPEC_V30) ||
63  vcp_version_eq(vspec, VCP_SPEC_V22);
64  return result;
65 }

68 // Note that MCCS (VCP) versioning forms a directed graph, not a linear ordering.
69 //
70 // The v3.0 spec is an extension of v2.1, not v2.2.
71 // Both v3.0 and v2.2 are successors to v2.1.
72 //
73 // -- v3.0
74 // |
75 // v1.0---v2.0---- v2.1
76 // |
77 // -- v2.2

80 /* Compares two version numbers.
81  *
82  * Aborts if an attempt is made to compare v2.2 with v3.0
83  * Note: Will require modification if a new spec appears
84  */
85 bool vcp_version_le(DDCA_MCCS_Version_Spec v1, DDCA_MCCS_Version_Spec v2) {
86  bool result = false;
87  assert( is_known_vcp_spec(v1) && is_known_vcp_spec(v2) );
88  assert( !(vcp_version_eq(v1, VCP_SPEC_V22) && vcp_version_eq(v2, VCP_SPEC_V30)) &&
89  !(vcp_version_eq(v2, VCP_SPEC_V22) && vcp_version_eq(v1, VCP_SPEC_V30))
90  );
91 
92  if (v1.major < v2.major)
93  result = true;
94  else if (v1.major == v1.minor) {
95  if (v1.minor <= v2.minor)
96  result = true;
97  }
98 
99  return result;
100 }

115 char * format_vspec(DDCA_MCCS_Version_Spec vspec) {
116  static char private_buffer[20];
117  if ( vcp_version_eq(vspec, VCP_SPEC_UNQUERIED) )
118  g_strlcpy(private_buffer, "Unqueried", sizeof(private_buffer)); // g_strlcpy() to quiet covrity
119  else if ( vcp_version_eq(vspec, VCP_SPEC_UNKNOWN) )
120  strcpy(private_buffer, "Unknown"); // will coverity flag this?
121  else
122  snprintf(private_buffer, 20, "%d.%d", vspec.major, vspec.minor);
123  return private_buffer;
124 }

159 DDCA_MCCS_Version_Spec parse_vspec(char * s) {
160  DDCA_MCCS_Version_Spec vspec;
161  int ct = sscanf(s, "%hhd . %hhd", &vspec.major, &vspec.minor);
162  if (ct != 2 || vspec.major > 3 || vspec.minor > 2) {
163  vspec = VCP_SPEC_UNKNOWN;
164  }
165  return vspec;

170 DDCA_MCCS_Version_Id mccs_version_spec_to_id(DDCA_MCCS_Version_Spec vspec) {
171  DDCA_MCCS_Version_Id result = DDCA_VUNK; // initialize to avoid compiler warning
172 
173  if (vspec.major == 1 && vspec.minor == 0)
174  result = DDCA_V10;
175  else if (vspec.major == 2 && vspec.minor == 0)
176  result = DDCA_V20;
177  else if (vspec.major == 2 && vspec.minor == 1)
178  result = DDCA_V21;
179  else if (vspec.major == 3 && vspec.minor == 0)
180  result = DDCA_V30;
181  else if (vspec.major == 2 && vspec.minor == 2)
182  result = DDCA_V22;
183  else if (vspec.major == 2 && vspec.minor == 1)
184  result = DDCA_V21;
185  else if (vspec.major == 0 && vspec.minor == 0)
186  result = DDCA_VUNK;
187  // case UNQUERIED should never arise
188  else
189  PROGRAM_LOGIC_ERROR("Unexpected version spec value %d.%d", vspec.major, vspec.minor);
190 
191  return result;
192 }
