244 Base_Status_ADL adl_ddc_set_vcp(int iAdapterIndex, int iDisplayIndex, Byte vcp_feature_code, int newval) {
245  if (adl_debug)
246  DBGMSG("Starting adapterNdx=%d, displayNdx=%d, vcp_feature_code=0x%02x",
247  iAdapterIndex, iDisplayIndex, vcp_feature_code );
248 
249  unsigned char ddc_cmd_bytes[] = {
250  0x6e, // address 0x37, shifted left 1 bit
251  0x51, // source address
252  0x02 | 0x04, // number of DDC data bytes, with high bit set
253  0x03, // DDC Set Feature Command
254  vcp_feature_code, //
255  newval & 0xff00 >> 8,
256  newval & 0xff,
257  0x00, // checksum, to be set
258  };
259  ddc_cmd_bytes[7] = ddc_checksum(ddc_cmd_bytes, 7, false); // calculate DDC checksum on all bytes
260 
261  int sendCt = sizeof(ddc_cmd_bytes);
262  assert (sendCt == 8);
263 
264  Base_Status_ADL rc = adl_ddc_write_only(iAdapterIndex, iDisplayIndex, ddc_cmd_bytes, sendCt);
